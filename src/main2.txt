#include <Arduino.h>
#include <U8g2lib.h>

const int echoPin = 5;
const int trigPin = 3;

bool toggle_var = true;
unsigned long lastTestTriggerTime = 0;

U8G2_SSD1306_128X64_NONAME_1_HW_I2C u8g2(U8G2_R0, /* reset=*/ 21, /* clock=*/ 18, /* data=*/ 17);

void setup() {
  Serial.begin(115200);
  
  pinMode(trigPin, OUTPUT);
  pinMode(echoPin, INPUT);

  pinMode(Vext, OUTPUT);
  digitalWrite(Vext, LOW);
  delay(100);
  
  u8g2.begin();
  u8g2.setFont(u8g2_font_6x10_tf);
}

void loop() {

  // STANDARD-Ablauf (alle 2000ms über das delay am Ende)
  toggle_var = !toggle_var;

  // Normaler Trigger
  digitalWrite(trigPin, LOW);
  delayMicroseconds(10);
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(2);
  digitalWrite(trigPin, LOW);

  // Echo messen
  long duration = pulseIn(echoPin, HIGH, 30000);
  long distance = duration * 0.034 / 2;

  // Display-Ausgabe
  u8g2.firstPage();
  do {
    u8g2.drawStr(0, 12, "Wassertank-Sensor");
    u8g2.drawHLine(0, 16, 128);

    if (duration == 0) {
      u8g2.setCursor(0, 35);
      u8g2.print("STATUS: TIMEOUT!");
      u8g2.setCursor(0, 55);
      u8g2.print("Sensor unplugged?");
    } else {
      u8g2.setCursor(0, 35);
      u8g2.print("STATUS: OKEY");
      
      u8g2.setCursor(0, 55);
      if (toggle_var == true) {
        u8g2.print("UltraSonicTOF: ");
        u8g2.print(duration);
        u8g2.print(" us");
      } else {
        u8g2.print("Distanz: ");
        u8g2.print(distance);
        u8g2.print(" cm");
      }
    } 
  } while ( u8g2.nextPage() );

  delay(2000); 
}
